HTML5拖放

demo:http://senguzh.com/zhuangtongfa/demo/drag.php
源码:

    <!DOCTYPE HTML>


    <html>


    <title>HTML5拖动demo</title>


    <head>


        <meta name="name" content="content" charset="utf-8">


        <style type="text/css">


            * {


                color: rgba(0, 0, 0, .6);


            }


            #div1 {


                width: 488px;


                height: 70px;


                padding: 10px;


                border: 1px solid rgba(0, 0, 0, .6);

            }

        </style>

        <script type="text/javascript">

            function allowDrop(ev) {

                ev.preventDefault();

            }

            function drag(ev) {

                ev.dataTransfer.setData("dom", ev.target.id);

            }

            function drop(ev) {

                ev.preventDefault();

                var data = ev.dataTransfer.getData("dom");

                ev.target.appendChild(document.getElementById(data));

            }

        </script>

    </head>

    <body>

        <div id="div1" ondrop="drop(event)" ondragover="allowDrop(event)"></div>

        <h1 id="hid" draggable="true" ondragstart="drag(event)">我是文字,请拖动我到框里</h1>

    </body>

    </html>
