简易柱形图demo

js控制css做的一个柱形图

demo:戳我
源码:


    <!DOCTYPE html>

&lt;html&gt;

&lt;head&gt;

&nbsp; &lt;meta charset=&quot;UTF-8&quot;&gt;

&nbsp; &lt;title&gt;柱形图&lt;/title&gt;

&lt;/head&gt;

&lt;body&gt;

&nbsp; &lt;style&gt;

&nbsp; &nbsp; #container {

&nbsp; &nbsp; &nbsp; height: 400px;

&nbsp; &nbsp; &nbsp; width: 400px;

&nbsp; &nbsp; &nbsp; background-color: transparent;

&nbsp; &nbsp; &nbsp; position: absolute;

&nbsp; &nbsp; &nbsp; left: 300px;

&nbsp; &nbsp; &nbsp; bottom: 30px;

&nbsp; &nbsp; }

&nbsp; &nbsp;&nbsp;

&nbsp; &nbsp; #box {

&nbsp; &nbsp; &nbsp; height: 30px;

&nbsp; &nbsp; &nbsp; width: 30px;

&nbsp; &nbsp; &nbsp; background-color: red;

&nbsp; &nbsp; &nbsp; position: absolute;

&nbsp; &nbsp; &nbsp; bottom: 0;

&nbsp; &nbsp; &nbsp; transition: 1s background-color;

&nbsp; &nbsp; }

&nbsp; &lt;/style&gt;

&nbsp; &lt;div id=&quot;container&quot;&gt;

&nbsp; &nbsp; &lt;div id=&quot;box&quot;&gt;

&nbsp; &nbsp; &nbsp; &lt;p&gt;&lt;/p&gt;

&nbsp; &nbsp; &lt;/div&gt;

&nbsp; &lt;/div&gt;

&nbsp; &lt;script type=&quot;text/javascript&quot;&gt;

&nbsp; &nbsp; (function() {

&nbsp; &nbsp; &nbsp; var div = document.getElementById(&#39;box&#39;);

&nbsp; &nbsp; &nbsp; var ss = setInterval(function() {

&nbsp; &nbsp; &nbsp; &nbsp; var h = getComputedStyle(div);

&nbsp; &nbsp; &nbsp; &nbsp; var hh = parseInt(h.height);

&nbsp; &nbsp; &nbsp; &nbsp; var p = div.getElementsByTagName(&#39;p&#39;)[0];

&nbsp; &nbsp; &nbsp; &nbsp; div.style.height = hh + 3 + &#39;px&#39;;

&nbsp; &nbsp; &nbsp; &nbsp; div.style.color = &#39;rgba(0,0,0,.5)&#39;;

&nbsp; &nbsp; &nbsp; &nbsp; p.innerHTML = parseInt(hh / 300 * 100 - 1) + &#39;%&#39;;

&nbsp; &nbsp; &nbsp; &nbsp; p.style.position = &quot;absolute&quot;;

&nbsp; &nbsp; &nbsp; &nbsp; p.style.bottom = &#39;-35px&#39;;


&nbsp; &nbsp; &nbsp; &nbsp; div.style.textAlign = &#39;center&#39;;

&nbsp; &nbsp; &nbsp; &nbsp; if (hh &gt; 100) {

&nbsp; &nbsp; &nbsp; &nbsp; &nbsp; div.style.backgroundColor = &quot;green&quot;;

&nbsp; &nbsp; &nbsp; &nbsp; }

&nbsp; &nbsp; &nbsp; &nbsp; if (hh &gt; 300) {

&nbsp; &nbsp; &nbsp; &nbsp; &nbsp; clearInterval(ss);

&nbsp; &nbsp; &nbsp; &nbsp; }

&nbsp; &nbsp; &nbsp; }, 5)

&nbsp; &nbsp; })()

&nbsp; &lt;/script&gt;

&lt;/body&gt;

&lt;/html&gt;


原理很简单,就是利用js控制css中的高度,只是加了个setInterval方法让高度慢慢变化
css动画局限性还是很大,js控制css就灵活多了,套用物理公式还可以模拟物理效果,在这里就不详细讲了
