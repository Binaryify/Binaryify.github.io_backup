简易柱形图demo

js控制css做的一个柱形图
源码:

    <!DOCTYPE html>

    <html>

    <head>

        <meta charset="UTF-8">

        <title>柱形图</title>

    </head>

    <body>

        <style>

            #container{

                height: 400px;

                width: 400px;

                background-color: transparent;

                position: absolute;

                left: 300px;

                bottom: 30px;

            }

            #box{

                height: 30px;

                width: 30px;

                background-color: red;

                position: absolute;

                bottom: 0;

                transition: 1s background-color;

                /*text-align: center;*/

            }


        </style>

    <div id="container"><div id="box"><p></p></div></div>

    <script type="text/javascript">

        (function(){

            var div=document.getElementById('box');

            // div.style.transs

            var ss=setInterval(function(){

                var h=getComputedStyle(div);

                var hh=parseInt(h.height);

                var p=div.getElementsByTagName('p')[0];

                div.style.height=hh+3+'px';

                // if(hh>70){

                //     div.style.backgroundColor="yellow";

                // }

                div.style.color='black';

                p.innerHTML=parseInt(hh/300*100-1)+'%';

                p.style.position="absolute";

                p.style.bottom='-35px';

                // div.style.lineHeight=hh+'px';

                div.style.textAlign='center';

                if(hh>100){

                    div.style.backgroundColor="green";

                }

                if(hh>300){

                    clearInterval(ss);

                }

            },5)

        })()

    </script>

    </body>

    </html>

css动画局限性还是很大,js控制css就灵活多了,套用物理公式还可以模拟物理效果
