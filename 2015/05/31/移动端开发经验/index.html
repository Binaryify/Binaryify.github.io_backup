    document.getElementById('div1').addEventListener("click",haha,false);

    function haha(ev){

        var ul=document.getElementById('div1').getElementsByTagName('ul')[0];

        var li=ul.getElementsByTagName('li');

        var liLen=li.length;

        var ul2=document.getElementById('container').getElementsByTagName('ul');

        

        

        for(i=0;i<liLen;i++){

            li[i].style.background="#eee";

            ul2[i].style.display="none";

            // console.log(ul2[i]);

        }

        // console.log(ev.target.nodeName.toLowerCase());

        ev.target.style.background="white";

        // console.log(ev.target.id);

        var id=ev.target.id;

        // console.log(id);

        ul2[id].style.display="block";

        

        

    }

    </script>
