关于创建svg圆形进度按钮的简单实现

发表于 2015-05-27

老大给我看了一个svg圆形进度按钮，ps：http://www.tuicool.com/articles/In6f2a ，让我照着这个做出来，觉得原来的太麻烦，各种库，好在前两天学了svg，自己弄了一早上，简单实现如下：
（由于是400%放大情况下做的，放大到400%才能正常显示，需要的可以再修改）


    <!DOCTYPE html>


    <html>

    <head>

    <meta charset="UTF-8">

    <title>Document</title>

    <body>

    <style type="text/css">

    .path1{

    stroke-width:2;

    }

    .path11 {

      stroke-dasharray: 500;

      stroke-dashoffset: 500;

      -webkit-animation: dash 3s linear forwards;

      stroke-width:2;

    }

    .path2 {

      stroke-dasharray: 28;

      stroke-dashoffset: 28;

      -webkit-animation: dash .3s linear forwards ;

     /* stroke-width:1;*/

    }

    .path3{

    stroke-dasharray: 13;

       stroke-dashoffset: 13;

      -webkit-animation: dash .3s linear forwards ;

    }

    @-webkit-keyframes dash {

      to {

        stroke-dashoffset: 0;

      }

    }

    .div1{

    width:250px;

    height:62px;

    border-radius:50px;

    border:2px solid green;

    -webkit-animation: myan .6s forwards;

    display:none;

    }

    /*.div1{

    -webkit-animation: myan .6s forwards;

    }*/

    #svgdiv{

    position:absolute;

    -webkit-transform: translate(128px,-68px);

    }

    @-webkit-keyframes myan{

    to {

    border:2px solid gray;

    border-radius:50%;

    width:62px;

    }

    }

    .ddd{

    border:2px solid gray;

    border-radius:50%;

    height:62px;

    width:62px;

    -webkit-animation: myam .6s forwards;

    }

    @-webkit-keyframes myam{

    to{

    width:250px;

    border-radius:50px;

    border:2px solid green;

    }

    }

    </style>

    document.getElementsByTagName('button')[0].addEventListener('click',haha,false);

    document.getElementsByTagName('button')[1].addEventListener('click',haha2,false);

    function haha(){

    document.getElementsByClassName('div1')[0].style.display="block";

    document.getElementById('pp').setAttribute("class","path11");

    setTimeout("haha3()",1700);

    // document.getElementById('pp').className='path11';

    // document.getElementById('pp2').className='path2';

    }

    function haha2(){

    document.getElementsByClassName('div1')[0].style.display="block";

    document.getElementById('pp').setAttribute("class","path11");

    setTimeout("haha4()",1700);

    }

    function haha4(){

    document.getElementById('pp').style.display="none";

    document.getElementById('pp3').style.display="inline";

    document.getElementById('pp3').setAttribute("class","path3");

    document.getElementsByClassName('div1')[0].className='ddd';

    // alert(tt);

    // document.getElementById('pp2').className='path2';

    }

    function haha3(){

    document.getElementById('pp').style.display="none";

    document.getElementById('pp2').style.display="inline";

    document.getElementById('pp2').setAttribute("class","path2");

    document.getElementsByClassName('div1')[0].className='ddd';

    }

    </script>
