微信上传下载多媒体php接口（转）


上传下载多媒体文件，这个作为高级接口的一部分功能，可能使用的人并不多，所以现在网上关于这个接口的使用教程很少。于是我就写了这个接口的使用方法，主要是两个函数




  1. //上传多媒体文件





  2.     function uploadMedia($url){





  3.         $file = realpath('1.mp3'); //要上传的文件





  4.         $fields['media'] = '@'.$file;





  5.         $ch = curl_init($url) ;





  6.         curl_setopt($ch, CURLOPT_POST, 1);





  7.         curl_setopt($ch, CURLOPT_CUSTOMREQUEST, "POST");





  8.         curl_setopt($ch, CURLOPT_POST, 1);





  9.         curl_setopt($ch, CURLOPT_POSTFIELDS,$fields);





  10.         curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);





  11.         $result = curl_exec($ch) ;





  12.         if (curl_errno($ch)) {





  13.          return curl_error($ch);





  14.         }





  15.         curl_close($ch);





  16.         return $result;





  17.     }










  19.  //下载多媒体文件





  20.     function saveMedia($url){





  21.         $ch = curl_init($url);





  22.         curl_setopt($ch, CURLOPT_HEADER, 0);    





  23.         curl_setopt($ch, CURLOPT_NOBODY, 0);    //对body进行输出。





  24.         curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);





  25.         $package = curl_exec($ch);





  26.         $httpinfo = curl_getinfo($ch);





  27.        





  28.         curl_close($ch);





  29.         $media = array_merge(array('mediaBody' => $package), $httpinfo);





  30.         





  31.         //求出文件格式





  32.         preg_match('/\w\/(\w+)/i', $media["content_type"], $extmatches);





  33.         $fileExt = $extmatches[1];





  34.         $filename = time().rand(100,999).".{$fileExt}";





  35.         $dirname = "./wximages/";





  36.         if(!file_exists($dirname)){





  37.             mkdir($dirname,0777,true);





  38.         }





  39.         file_put_contents($dirname.$filename,$media['mediaBody']);





  40.         return $dirname.$filename;





  41.     }





复制代码

函数中的url就是官方说明文档中的url，然后跟上相关参数即可
