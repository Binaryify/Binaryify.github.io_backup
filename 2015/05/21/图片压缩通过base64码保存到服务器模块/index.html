图片压缩通过base64码保存到服务器模块

前台代码：

         <script src="lrz.mobile.min.js"></script>
    <input type="file" >

    <form action="1.php" method="post">

        <input type="text" name="bb">

        <input type="submit" value="提交">

    </form>

        <script>

    var input = document.querySelector('input');

    input.onchange = function () {

        // 也可以传入图片路径：lrz('../demo.jpg', ...

        lrz(this.files[0], function (results) {

          // 你需要的数据都在这里，可以以字符串的形式传送base64给服务端转存为图片。

          document.getElementsByTagName('p')[0].innerHTML=results.base64;

          document.getElementsByTagName('input')[1].value=results.base64;

        });

    }

        </script>

后台代码：
